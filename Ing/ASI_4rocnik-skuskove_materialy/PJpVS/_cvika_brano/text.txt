void naplnPole(int pole[], int n, int a, int b){
     for (int i = 0; i < n; i++){
        pole[i] = 10 + rand() %(b - a + 1);  //random cisla od 0 - b - a + 1 (horna hranica - dolna + 1))
    }
}

void vypisPole(int pole[], int n){
    for(int i = 0; i < n; i++){
        printf("%d ", pole[i]);
    }
    printf("\n");
}

void zmenVelkostPola(int *pole[], int staraVelkost, int novaVelkost){
    if(novaVelkost > staraVelkost){
        int *novePole = (int *)calloc(novaVelkost, sizeof(int));
        for(int i = 0; i < staraVelkost; i++){
            novePole[i] = (*pole)[i];
        }
        free(*pole);
        *pole = novePole;
        free(novePole);
    }
}

void vymenSmernikyPoli(int *pole1[], int *pole2[]){
    int *pomPole = *pole1;
    *pole1 = *pole2;
    *pole2 = pomPole;
}
